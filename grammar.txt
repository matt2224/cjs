root_defs -> function_def root_defs
           | enum_def root_defs
           | <epsilon>
enum_def -> enum { enum_list_item enum_list_cont }
enum_list_item -> id
enum_list_cont -> , enum_list_item enum_list_cont
               | <epsilon>
function_def -> type id ( arg_list ) { stmt_list }
args_list -> args_list_item args_list_cont
           | <epsilon>
args_list_cont -> , args_list_item args_list_cont
               | <epsilon>
args_list_item -> type id
stmt_list -> stmt ; stmt_list
             | <epsilon>
stmt -> return literal
      | type id = literal
literal -> num
        | string
type -> int | char | void
